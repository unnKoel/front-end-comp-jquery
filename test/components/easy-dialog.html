easyDialog 弹出层演示

默认状态

            easyDialog.open({
                container : 'testBox'
            });
run

不显示遮罩层

            easyDialog.open({
              container : 'testBox',
              isOverlay : false
            });
run

弹出窗口绝对定位

            easyDialog.open({
              container : 'testBox',
              fixed : false
            });
run

弹出窗口跟随页面某元素定位

            easyDialog.open({
              container : 'testBox',
              follow : 'demoBtn4',
              followX : -265,
              followY : 24
            });
run

自动关闭窗口

            easyDialog.open({
              container : 'testBox',
              autoClose : 1500
            });
run

关闭窗口的回调函数

            easyDialog.open({
              container : 'testBox',
              callback : message
            });

            function message(){
              alert('hello');
            };
run

关闭窗口

            easyDialog.close();
